30
GitHub Actions CI/CD Pipeline
Predecessor: Step 29 Successor: Step 31 Owner: DevOps Engineer Time: 1 day 
# .github/workflows/deploy.yml
name: deploy
on: { push: { branches: [main] } }
jobs:
  build-test-deploy:
    steps:
      - run: pip install -r requirements.txt && pytest
      - run: docker build -t $ECR_URI:$GITHUB_SHA .
      - run: aws ecr get-login-password | docker login ... && docker push $ECR_URI:$GITHUB_SHA
      - run: kubectl set image deployment/ai-me-bank-api api=$ECR_URI:$GITHUB_SHA -n ai-me-bank
      - run: kubectl rollout status deployment/ai-me-bank-api -n ai-me-bank
      - run: ./scripts/smoke_test.sh   # Step 31
    # Production deploys gated on a tagged release, not every main-branch push

CloudWatch Alarms + SNS
Predecessor: Step 31 (Gate 2 passed) Successor: Step 33 (Part 4) Owner: DevOps Engineer Time: 1 hour 
aws cloudwatch put-metric-alarm --alarm-name ai-me-bank-high-cpu \
  --metric-name CPUUtilization --namespace AWS/EKS --threshold 80 \
  --comparison-operator GreaterThanThreshold --evaluation-periods 3 \
  --alarm-actions $SNS_TOPIC_ARN --profile ai-me-bank-prod

aws sns subscribe --topic-arn $SNS_TOPIC_ARN --protocol email --notification-endpoint devops-lead@yourbank.com
